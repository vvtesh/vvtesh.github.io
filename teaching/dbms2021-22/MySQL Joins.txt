SELECT student_info.f_name, student_info.course_ID 
FROM student_info 
INNER JOIN courses
ON student_info.course_ID = courses.course_ID;

SELECT *
FROM student_info 
LEFT JOIN courses
ON student_info.course_ID = courses.course_ID;

SELECT *
FROM student_info 
RIGHT JOIN courses
ON student_info.course_ID = courses.course_ID;

SELECT a.roll_num , b.f_name
FROM students a, students b
WHERE a.roll_num < b.roll_num;

SELECT * FROM student_info CROSS JOIN courses;

create table prereq (course_id varchar(20), prereq_id varchar(20));

insert into prereq values ('BIO-301', 'BIO-101');
insert into prereq values ('CS-190', 'CS-101');
insert into prereq values ('CS-347', 'CS-101');

SELECT *
FROM course 
CROSS JOIN prereq;


SELECT c.course_id, p.prereq_id, c.title, c.dept_name, c.credits FROM course c LEFT JOIN
prereq p USING(course_id)
UNION
SELECT p.course_id, p.prereq_id, c.title, c.dept_name, c.credits FROM course c RIGHT
JOIN prereq p USING(course_id);
